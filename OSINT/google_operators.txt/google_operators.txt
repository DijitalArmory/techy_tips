# Search in specific site
# Example:
# <expression> site: site.com
# <expression> AND <expression> site: site.com

# Search for specific string
# Example:
# "<string>" 
# "<string>" site: site.com
# "<string>" AND "<string>" site: site.com

# Search with wildcard
# Example:
# "The * mentor" 

# Search for filetype
# Example:
# site: site.com password filetype: pdf

# Search for site subdomains
# Example:
# site: site.com -www

# Exclude an object
# Example:
# site: site.com -www -forums

# Example:
# <subject> intext: <password>

# Example:
# <subject> inurl: <text> site: site.com
